export const handleApiSuccess = (notify, message) => {
  notify.success("Thành công", message);
};

export const handleApiError = (error, form, notify) => {
  const err = error?.error || error;

  // ✅ VALIDATION (ZOD / backend trả về)
  if (err?.code === "VALIDATION_ERROR" && err?.errors) {
    const formErrors = err.errors.map((e) => ({
      name: e.field,
      errors: [e.message],
    }));

    form?.setFields(formErrors);
    return;
  }

  // ❌ BUSINESS ERROR
  notify.error("Thất bại", err?.message || "Có lỗi xảy ra");
};
